-- Step 1: Create Employee Table
CREATE TABLE Employee (
    id NUMBER PRIMARY KEY,
    name VARCHAR2(50),
    salary NUMBER
);

---------------------------------------------------
-- Step 4: Create Stored Procedure using Cursor
---------------------------------------------------

CREATE OR REPLACE PROCEDURE Increase_Salary
AS
    -- Declare variables
    v_id Employee.id%TYPE;

    -- Declare cursor
    CURSOR emp_cursor IS
        SELECT id FROM Employee;

BEGIN
    -- Open cursor
    OPEN emp_cursor;

    LOOP
        -- Fetch employee id
        FETCH emp_cursor INTO v_id;

        -- Exit when no records left
        EXIT WHEN emp_cursor%NOTFOUND;

        -- Increase salary by 1000
        UPDATE Employee
        SET salary = salary + 1000
        WHERE id = v_id;

    END LOOP;

    -- Close cursor
    CLOSE emp_cursor;

    COMMIT;

    DBMS_OUTPUT.PUT_LINE('Salary Updated Successfully');

END;
/

---------------------------------------------------
-- Step 5: Execute Procedure
---------------------------------------------------

BEGIN
    Increase_Salary;
END;
/
