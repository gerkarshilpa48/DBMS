    student_id INT,
    student_name VARCHAR(100),
    subject VARCHAR(50),
    marks INT
);

SELECT 
    MAX(marks) AS highest_marks,
    MIN(marks) AS lowest_marks,
    AVG(marks) AS average_marks
FROM Marks;

SELECT 
    subject,
    MAX(marks) AS highest_marks,
    MIN(marks) AS lowest_marks,
    AVG(marks) AS average_marks
FROM Marks
GROUP BY subject;


SELECT 
    subject,
    COUNT(student_id) AS total_students
FROM Marks
GROUP BY subject;


SELECT 
    subject,
    AVG(marks) AS average_marks
FROM Marks
GROUP BY subject
HAVING AVG(marks) > 70;
