-- Step 1: Create Bank Table
---------------------------------------------------

CREATE TABLE Bank (
    acc_id NUMBER PRIMARY KEY,
    name VARCHAR2(50),
    balance NUMBER
);

---------------------------------------------------
-- Step 3: Trigger to Set NULL Balance to 0
---------------------------------------------------

CREATE OR REPLACE TRIGGER trg_set_balance
BEFORE INSERT OR UPDATE
ON Bank
FOR EACH ROW
BEGIN
    IF :NEW.balance IS NULL THEN
        :NEW.balance := 0;
    END IF;
END;
/

---------------------------------------------------
-- Step 4: Trigger to Prevent Negative Balance
---------------------------------------------------

CREATE OR REPLACE TRIGGER trg_negative_balance
BEFORE INSERT OR UPDATE
ON Bank
FOR EACH ROW
BEGIN
    IF :NEW.balance < 0 THEN
        RAISE_APPLICATION_ERROR(
            -20001,
            'Balance cannot be negative'
        );
    END IF;
END;
/
